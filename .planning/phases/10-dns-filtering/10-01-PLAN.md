   ```kotlin
   fun toggleDnssec(enabled: Boolean) {
       viewModelScope.launch {
           _uiState.update { it.copy(isTogglingFilter = true) }
           val update = SecurityConfigUpdateDto(dnssecEnabled = enabled)
           securityRepository.updateSecurityConfig(update)
               .onSuccess { updated ->
                   _uiState.update { it.copy(isTogglingFilter = false, securityConfig = updated, actionMessage = "DNSSEC ${if(enabled) "aktif" else "pasif"}") }
               }
               .onFailure { e ->
                   _uiState.update { it.copy(isTogglingFilter = false, actionMessage = "Hata: ${e.message}") }
               }
       }
   }
   ```

   Constructor: `class DnsFilteringViewModel(private val securityRepository: SecurityRepository) : ViewModel()`
