   private fun formatBps(bps: Long): String {
       return when {
           bps >= 1_000_000_000 -> String.format("%.1f Gbps", bps / 1_000_000_000.0)
           bps >= 1_000_000 -> String.format("%.1f Mbps", bps / 1_000_000.0)
           bps >= 1_000 -> String.format("%.1f Kbps", bps / 1_000.0)
           else -> "$bps bps"
       }
   }
